function solution(N, stages) {
    stages.sort((a, b) => a - b)

    let tem = []
    for(let i = 1; i < N+1; i++){
        let fail = stages.filter(v => v === i).length / stages.length
        stages = stages.slice(stages.filter(v => v === i).length)
        tem.push([fail,i])
    }

    tem.sort((a,b) => {
        if(a[0]===b[0]) {return a[1]-b[1]}
        return b[0] - a[0]
    })

    return tem.map(v => v[1])
}
